```python
def calculate_radar_scores(dealer_code: str, year: int, month: int) -> dict:
    """
    计算五力雷达得分
    
    五力模型：
    1. 传播获客力 = 客流量 + 潜客量 + 线索量
    2. 体验力 = 试驾数
    3. 转化力 = 成交率 - 战败率
    4. 服务力 = 成交响应时间 + 战败响应时间
    5. 经营力 = 销量 + 政策 + GSEV
    """
    source = RadarSource2024.query.filter_by(dealer_code=dealer_code).first()
    
    month_field = f'_m{month:02d}'
    
    # 传播获客力
    spread_force = (
        getattr(source, f'customer_flow{month_field}', 0) or 0 +
        getattr(source, f'potential_customers{month_field}', 0) or 0 +
        getattr(source, f'leads{month_field}', 0) or 0
    )
    
    # 体验力
    experience_force = getattr(source, f'test_drives{month_field}', 0) or 0
    
    # 转化力
    success_rate = getattr(source, f'success_rate{month_field}', 0) or 0
    defeat_rate = getattr(source, f'defeat_rate{month_field}', 0) or 0
    conversion_force = success_rate - defeat_rate
    
    # 服务力
    service_force = (
        getattr(source, f'success_response_time{month_field}', 0) or 0 +
        getattr(source, f'defeat_response_time{month_field}', 0) or 0
    )
    
    # 经营力
    operation_force = (
        getattr(source, f'sales{month_field}', 0) or 0 +
        getattr(source, f'policy{month_field}', 0) or 0 +
        getattr(source, f'gsev{month_field}', 0) or 0
    )
    
    return {
        'spread_force': spread_force,
        'experience_force': experience_force,
        'conversion_force': conversion_force,
        'service_force': service_force,
        'operation_force': operation_force,
    }
```
